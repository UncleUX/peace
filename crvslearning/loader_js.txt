// JavaScript pour le loader terminal
document.addEventListener('DOMContentLoaded', function() {
    const loader = document.getElementById('terminalLoader');
    const progressBar = document.querySelector('.terminal-progress-bar');
    const progressText = document.querySelector('.terminal-progress-text');
    
    // Animation de la barre de progression
    let progress = 0;
    const progressInterval = setInterval(() => {
        progress += Math.random() * 15;
        if (progress > 100) progress = 100;
        
        progressBar.style.width = progress + '%';
        progressText.textContent = Math.floor(progress) + '%';
        
        if (progress >= 100) {
            clearInterval(progressInterval);
            setTimeout(() => {
                loader.style.opacity = '0';
                setTimeout(() => {
                    loader.style.display = 'none';
                }, 500);
            }, 500);
        }
    }, 200);
    
    // Ajouter des lignes de log dynamiques
    const terminalBody = document.querySelector('.terminal-body');
    const additionalLogs = [
        '✓ User permissions verified',
        '✓ Cache initialized',
        '✓ Security protocols active',
        '✓ Dashboard components ready'
    ];
    
    let logIndex = 0;
    const logInterval = setInterval(() => {
        if (logIndex < additionalLogs.length && progress < 80) {
            const newLine = document.createElement('div');
            newLine.className = 'terminal-line';
            newLine.style.opacity = '0';
            newLine.innerHTML = `<span class="terminal-output">${additionalLogs[logIndex]}</span>`;
            terminalBody.insertBefore(newLine, terminalBody.querySelector('.terminal-progress'));
            
            setTimeout(() => newLine.style.opacity = '1', 100);
            logIndex++;
        } else {
            clearInterval(logInterval);
        }
    }, 600);
});
